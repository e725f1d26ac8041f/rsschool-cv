    typeof sampleActivity === 'string' &&
    isFinite(sampleActivity) &&
    sampleActivity > 0 &&

  ) {
    const sampleAge = Math.log(MODERN_ACTIVITY / sampleActivity) / RADIOACTIVE_DECAY_CONSTANT;
